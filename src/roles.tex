% "Repeat n times" code from https://tex.stackexchange.com/questions/300207/repeat-characters-n-times
\makeatletter
\newcount\my@repeat@count
\newcommand{\myrepeat}[2]{%
  \begingroup
  \my@repeat@count=\z@
  \@whilenum\my@repeat@count<#1\do{#2\advance\my@repeat@count\@ne}%
  \endgroup
}
\makeatother

    \begin{centering}
    \ifnum0=#2 \newcommand{\xp}{2} 
    \else \ifnum1=#2 \newcommand{\xp}{3}
      \else \ifnum2=#2 \newcommand{\xp}{4}
        \else \newcommand{\xp}{0}
        \fi
      \fi
    \fi

    \setcounter{Level}{#2}
          
    \textbf{\large{#1}}\\
    Level #2\\
  XP: \ifnum3=#2 \else \myrepeat{\xp}{$\square$} \fi
\end{centering}
